/**
 * @file TodoModel - Core business logic and data management for the Todo app.
 * @description Manages todo items, handles persistence via the provided StorageService,
 * and implements a simple observer pattern for reactive UI updates.
 */

/**
 * @typedef {Object} Todo
 * @property {number} id - Unique identifier for the todo.
 * @property {string} text - Description text for the task.
 * @property {boolean} completed - Indicates whether the task is done.
 * @property {string} createdAt - ISO timestamp when the todo was created.
 */

/**
 * @callback Listener
 * @description Function called whenever the model updates.
 */

/**
 * @class TodoModel
 * @classdesc Manages todos, persistence, and subscriptions.
 */
export class TodoModel {
  /**
   * @param {*} storageService Storage engine that provides `save`, `load`, and `remove` methods.
   */
  constructor(storageService) {
    /** @private */
    this.storage = storageService;
    /** @type {Todo[]} */
    this.todos = this.storage.load('items', []);
    /** @type {Listener[]} */
    this.listeners = [];
    /** @type {number} */
    this.nextId = this.storage.load('nextId', 1);
  }

  /**
   * Subscribe to model changes.
  * @param {Listener} listener - Callback to invoke when data changes.
  * @returns {Function} Function to unsubscribe.
   */
  subscribe(listener) {
    if (typeof listener !== 'function') return () => {};
    if (this.listeners.indexOf(listener) === -1) this.listeners.push(listener);
    return () => {
      this.listeners = this.listeners.filter(l => l !== listener);
    };
  }

  /**
   * Notify all subscribers of model updates.
  * @private
  * @returns {void}
   */
  notify() {
    const list = Array.from(this.listeners);
    list.forEach(listener => {
      try {
        listener();
      } catch (e) {
        console.error('Listener error in TodoModel.notify:', e);
      }
    });
  }

  /**
   * Add a new todo.
   * @param {string} text - Task text (trimmed before saving).
  * @returns {Todo|null} The created todo or `null` if input invalid.
   */
  addTodo(text) {
    if (!text || (typeof text === 'string' && text.trim() === '')) return null;
    const txt = typeof text === 'string' ? text.trim() : String(text);
    const todo = {
      id: this.nextId++,
      text: txt,
      completed: false,
      createdAt: new Date().toISOString(),
    };
    this.todos.push(todo);
    this.save();
    this.notify();
    return todo;
  }

  /**
   * Toggle a todo's completion status.
   * @param {number} id - Todo identifier.
   * @returns {void}
   */
  toggleComplete(id) {
    const todo = this.todos.find(t => t.id === id);
    if (todo) {
      todo.completed = !todo.completed;
      this.save();
      this.notify();
    }
  }

  /**
   * Backwards-compatible alias for `toggleComplete`.
   * @param {number} id
   * @returns {void}
   */
  toggleTodo(id) {
    return this.toggleComplete(id);
  }

  /**
   * Delete a todo by its ID.
   * @param {number} id
   * @returns {void}
   */
  deleteTodo(id) {
    this.todos = this.todos.filter(t => t.id !== id);
    this.save();
    this.notify();
  }

  /**
   * Update a todo's text.
   * @param {number} id - Todo identifier.
   * @param {string|{text:string}} newText - New text value or object with `text`.
   * @returns {void}
   */
  updateTodo(id, newText) {
    const todo = this.todos.find(t => t.id === id);
    if (!todo) return;
    let text = null;
    if (typeof newText === 'string') text = newText;
    else if (newText && typeof newText === 'object' && 'text' in newText)
      text = newText.text;
    if (text && text.trim() !== '') {
      todo.text = text.trim();
      this.save();
      this.notify();
    }
  }

  /**
   * Clear all completed todos.
   * @returns {void}
   */
  clearCompleted() {
    this.todos = this.todos.filter(t => !t.completed);
    this.save();
    this.notify();
  }

  /**
   * Remove all todos (useful for reset or testing).
   * @returns {void}
   */
  clearAll() {
    this.todos = [];
    this.save();
    this.notify();
  }

  /**
   * Number of active (not completed) todos.
   * @type {number}
   * @readonly
   */
  get activeCount() {
    return this.todos.filter(t => !t.completed).length;
  }

  /**
   * Number of completed todos.
   * @type {number}
   * @readonly
   */
  get completedCount() {
    return this.todos.filter(t => t.completed).length;
  }

  /**
   * Persist todos and counter to storage.
   * @returns {void}
   */
  save() {
    this.storage.save('items', this.todos);
    this.storage.save('nextId', this.nextId);
  }
}
