 * @file StorageService - localStorage wrapper for the TODO app.
 * Persists JSON-serializable data under a namespaced key.
 */

/**
 * JSON-serializable value (kept loose for JSDoc compatibility).
 * @typedef {any} JSONValue
 */

/**
 * Simple key/value storage wrapper around `localStorage`.
 * Keys are automatically prefixed with `storageKey`.
 */
export class StorageService {
  /**
   * @param {string} [storageKey='lab9_todos'] - Prefix used for keys in localStorage.
   */
  constructor(storageKey = 'lab9_todos') {
    /** @private */
    this.storageKey = storageKey;
  }

  /**
   * Compute the fully-qualified key with the storage prefix.
   * @private
   * @param {string} key
   * @returns {string}
   */
  _full(key) {
    return `${this.storageKey}_${key}`;
  }

  /**
   * Save data to localStorage (JSON-serialized).
   * @param {string} key - Unprefixed logical key (e.g., "items").
   * @param {JSONValue} data - JSON-serializable data to store.
   * @returns {void}
   */
  save(key, data) {
    try {
      localStorage.setItem(this._full(key), JSON.stringify(data));
    } catch (error) {
      // eslint-disable-next-line no-console
      console.error('Failed to save to localStorage:', error);
    }
  }

  /**
   * Load data from localStorage (JSON-parsed).
   * @template T
   * @param {string} key - Unprefixed logical key (e.g., "items").
   * @param {T} [defaultValue=null] - Value to return when key is missing or parse fails.
   * @returns {T} Parsed value or the provided default.
   */
  load(key, defaultValue = /** @type {any} */ (null)) {
    try {
      const raw = localStorage.getItem(this._full(key));
      return raw ? JSON.parse(raw) : defaultValue;
    } catch (error) {
      // eslint-disable-next-line no-console
      console.error('Failed to load from localStorage:', error);
      return defaultValue;
    }
  }

  /**
   * Remove a single key from localStorage.
   * @param {string} key - Unprefixed logical key.
   * @returns {void}
   */
  remove(key) {
    try {
      localStorage.removeItem(this._full(key));
    } catch (error) {
      // eslint-disable-next-line no-console
      console.error('Failed to remove from localStorage:', error);
    }
  }

  /**
   * Clear **all** entries belonging to this app's namespace (matching the prefix).
   * @returns {void}
   */
  clear() {
    try {
      const toRemove = [];
      for (let i = 0; i < localStorage.length; i++) {
        const k = localStorage.key(i);
        if (k && k.startsWith(this.storageKey)) toRemove.push(k);
      }
      toRemove.forEach(k => localStorage.removeItem(k));
    } catch (error) {
      // eslint-disable-next-line no-console
      console.error('Failed to clear localStorage:', error);
    }
  }

  /**
   * Alias for {@link clear}.
   * @returns {void}
   */
  clearAll() {
    this.clear();
  }
}
